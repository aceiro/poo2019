RESPOSTA 4 - A STL implementa alguma funções usando o esquema de container e temaplates. Com ela podemos usar estruturas
como list, queue, vectors, stack, etc, dentre outras ferramentas. Para utilizar, devemos incluir a biblioteca no cabeçalho
do código.

Exemplo: usando o vector

#include <iostream> 
#include <vector> 

using namespace std; 

int main() 
{ 
    vector<int> g1; 

    for (int i = 1; i <= 20; i++) 
        g1.push_back(i); 

    cout << "Size : " << g1.size(); 
    cout << "\nCapacity : " << g1.capacity(); 
    cout << "\nMax_Size : " << g1.max_size(); 

    // redimensiona o vetor para tamanho 4 
    g1.resize(10); 

    // imprime o tamanho do vetor após o resize() 
    cout << "\nSize : " << g1.size(); 

    // verifica se o vetor não está vazio
    if (g1.empty() == false) 
        cout << "\nVector is not empty"; 
    else
        cout << "\nVector is empty"; 

    // Shrinks the vector 
    g1.shrink_to_fit(); 
    cout << "\nVector elements are: "; 
    for (auto it = g1.begin(); it != g1.end(); it++) 
        cout << *it << " "; 

    return 0; 
}
